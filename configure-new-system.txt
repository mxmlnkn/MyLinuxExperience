    mkdir -p /media/f/Linux/backups

	# .config/{Bitcoin,bitpay,Digibyte,Exodus,Jaxx,Litecoin,Mist,NanoWallet,PIVX,Reddcoin,Sia-UI}

    sudo tar --format=posix --absolute-names -cjpf "/media/f/Linux/backups/etc-$( hostname )-$( date +%Y-%m-%dT%H-%M-%S ).tar.bz2" /etc/apt/{preferences,apt.conf.d/30norecommends,sources.list} /etc/cron.daily/autocommit /etc/cron.weekly/backuphypatiahome /etc/fail2ban/ /etc/X11/xorg.conf.d/50-synaptics.conf /usr/share/images/desktop-base/lain-red-sitting-1920x1080.png /etc/lightdm/lightdm-gtk-greeter.conf /etc/ssh/ /root/ /usr/share/images/desktop-base/desktop-grub.png /etc/wpa_supplicant/TUD-CACert.pem

    tar --format=posix -cjf "/media/f/Linux/backups/home-$( hostname )-personal-settings-$( date +%Y-%m-%dT%H-%M-%S ).tar.bz2" .thunderbird/ .purple/ .local/share/TelegramDesktop/ .config/opera

# --overwrite not necessary, it should be default behavior
sudo chmod a+w /media
sudo chmod a+r /var/log/{syslog,auth.log}
tar --atime-preserve --overwrite -xvf home-hypatia-pc-settings.tar
sudo tar --atime-preserve --absolute-names -xvf --overwrite opt.tar

tar --atime-preserve -xf home-LenovoE330-personal-settings.tar.bz2
ln -s /media/f/Linux/bin ~/bin
ln -s /media/f/Linux/etc ~/etc
# reboot -> WUNDERBAR. sieht mit einem mal schon sehr stark nach meinem PC aus :333

sudo chmod a+r /var/log/{syslog,auth.log}   # for conky to work
sudo update-grub # apply lain wallpaper
/media/f/Linux/install-packets.sh
si libxfce4ui-1-dev libexif-dev && cd /opt/ristretto-0.8.2. && ./configure && sudo make install
